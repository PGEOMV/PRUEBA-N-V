    PROCEDURE AmountCalculation@1120020012();
    VAR
      Cabecera@1120020004 : Code[20];
      RegCabeceraFac@1120020003 : Record 112;
      RecCabeceraAbo@1120020002 : Record 114;
      CamposFacturae@1120020001 : Record 57010;
      Tipo@1120020000 : Text[30];
    BEGIN
      CamposFacturae.RESET;
      CamposFacturae.SETRANGE("A�O FACTURA", 2014);
      //CamposFacturae.SETRANGE("IMPORTE NETO",0);
      IF CamposFacturae.FIND('-') THEN
         REPEAT
            RegCabeceraFac.RESET;
            RegCabeceraFac.SETRANGE("No.", CamposFacturae."No. Factura");
            IF RegCabeceraFac.FIND('-') THEN
               BEGIN
                  RegCabeceraFac.CALCFIELDS(Amount, "Amount Including VAT");
                  CamposFacturae."IMPORTE NETO" := RegCabeceraFac.Amount;
                  CamposFacturae."IMPORTE IVA" :=  RegCabeceraFac."Amount Including VAT" - RegCabeceraFac.Amount;
                  CamposFacturae."TOTAL FACTURA" := RegCabeceraFac."Amount Including VAT";
                  CamposFacturae.MODIFY;
               END;

         UNTIL CamposFacturae.NEXT = 0;


                  {
                  IF Tipo = 'F' THEN
                     BEGIN
                      //  RegCabeceraFac.SETRANGE("No.", Cabecera);
                        RegCabeceraFac.GET(Cabecera);
                      //  IF RegCabeceraFac.FIND('-') THEN
                           BEGIN
                               RegCabeceraFac.CALCFIELDS(Amount, "Amount Including VAT");
                               CamposFacturae.RESET;
                               CamposFacturae.SETRANGE("No. Factura", Cabecera);
                               IF CamposFacturae.FIND('-') THEN
                                  BEGIN
                                     CamposFacturae."IMPORTE NETO" := RegCabeceraFac.Amount;
                                     CamposFacturae."IMPORTE IVA" :=  RegCabeceraFac."Amount Including VAT" - RegCabeceraFac.Amount;
                                     CamposFacturae.MODIFY;
                                  END
                           END;
                     END;
                  IF Tipo = 'A' THEN
                     BEGIN
                        RecCabeceraAbo.SETRANGE("No.", Cabecera);
                        IF RecCabeceraAbo.FIND('-') THEN
                           BEGIN
                               RecCabeceraAbo.CALCFIELDS(Amount, "Amount Including VAT");
                               CamposFacturae.RESET;
                               CamposFacturae.SETRANGE("No. Factura", Cabecera);
                               IF CamposFacturae.FIND('-') THEN
                                  BEGIN
                                     CamposFacturae."IMPORTE NETO" := RecCabeceraAbo.Amount;
                                     CamposFacturae."IMPORTE IVA" :=  RecCabeceraAbo."Amount Including VAT" - RegCabeceraFac.Amount;
                                     CamposFacturae.MODIFY;
                                  END
                           END;
                     END;
                   }

      CabVenta.RESET;
      CabVenta.SETFILTER("No.", '<>%1', '');
      IF CabVenta.FIND('-') THEN
         CabVenta.DELETEALL;
